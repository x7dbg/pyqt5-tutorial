# 多线程编程

在 GUI 应用中，耗时操作会阻塞主线程，导致界面无响应。多线程可以解决这个问题。

---

## 1. 为什么需要多线程

### 1.1 单线程的问题

```python
# -*- coding: utf-8 -*-
import sys
import time
from PyQt5.QtWidgets import QApplication, QWidget, QPushButton, QTextEdit, QVBoxLayout

class Example(QWidget):
    def __init__(self):
        super().__init__()
        self.initUI()
    
    def initUI(self):
        self.textEdit = QTextEdit()
        self.btn = QPushButton('开始任务')
        self.btn.clicked.connect(self.longTask)
        
        layout = QVBoxLayout()
        layout.addWidget(self.textEdit)
        layout.addWidget(self.btn)
        self.setLayout(layout)
        
        self.setWindowTitle('单线程示例')
        self.show()
    
    def longTask(self):
        self.textEdit.append('开始任务...')
        time.sleep(3)  # 模拟长时间任务
        self.textEdit.append('任务完成')

if __name__ == '__main__':
    app = QApplication(sys.argv)
    ex = Example()
    sys.exit(app.exec_())
```

点击按钮后，界面会卡住 3 秒，期间无法进行任何操作。

## 2. QThread 基础用法

### 2.1 创建线程

```python
# -*- coding: utf-8 -*-
import sys
import time
from PyQt5.QtCore import QThread, pyqtSignal
from PyQt5.QtWidgets import QApplication, QWidget, QPushButton, QTextEdit, QVBoxLayout

class Worker(QThread):
    finished = pyqtSignal(str)
    
    def run(self):
        # 在线程中执行长时间任务
        time.sleep(3)
        self.finished.emit('任务完成')

class Example(QWidget):
    def __init__(self):
        super().__init__()
        self.initUI()
    
    def initUI(self):
        self.textEdit = QTextEdit()
        self.btn = QPushButton('开始任务')
        self.btn.clicked.connect(self.startTask)
        
        layout = QVBoxLayout()
        layout.addWidget(self.textEdit)
        layout.addWidget(self.btn)
        self.setLayout(layout)
        
        self.setWindowTitle('多线程示例')
        self.show()
    
    def startTask(self):
        self.btn.setEnabled(False)
        self.textEdit.append('开始任务...')
        
        # 创建并启动线程
        self.worker = Worker()
        self.worker.finished.connect(self.onFinished)
        self.worker.start()
    
    def onFinished(self, message):
        self.textEdit.append(message)
        self.btn.setEnabled(True)

if __name__ == '__main__':
    app = QApplication(sys.argv)
    ex = Example()
    sys.exit(app.exec_())
```

## 3. 带进度的线程

```python
# -*- coding: utf-8 -*-
import sys
import time
from PyQt5.QtCore import QThread, pyqtSignal
from PyQt5.QtWidgets import (QApplication, QWidget, QPushButton, 
                             QTextEdit, QVBoxLayout, QProgressBar)

class Worker(QThread):
    progress = pyqtSignal(int)
    finished = pyqtSignal(str)
    
    def run(self):
        for i in range(1, 101):
            time.sleep(0.05)
            self.progress.emit(i)
        self.finished.emit('任务完成')

class Example(QWidget):
    def __init__(self):
        super().__init__()
        self.initUI()
    
    def initUI(self):
        self.textEdit = QTextEdit()
        self.progressBar = QProgressBar()
        self.btn = QPushButton('开始任务')
        self.btn.clicked.connect(self.startTask)
        
        layout = QVBoxLayout()
        layout.addWidget(self.textEdit)
        layout.addWidget(self.progressBar)
        layout.addWidget(self.btn)
        self.setLayout(layout)
        
        self.setWindowTitle('带进度的多线程')
        self.show()
    
    def startTask(self):
        self.btn.setEnabled(False)
        self.textEdit.append('开始任务...')
        self.progressBar.setValue(0)
        
        self.worker = Worker()
        self.worker.progress.connect(self.progressBar.setValue)
        self.worker.finished.connect(self.onFinished)
        self.worker.start()
    
    def onFinished(self, message):
        self.textEdit.append(message)
        self.btn.setEnabled(True)

if __name__ == '__main__':
    app = QApplication(sys.argv)
    ex = Example()
    sys.exit(app.exec_())
```

## 4. 线程池 QThreadPool

对于大量短任务，使用线程池更高效。

```python
# -*- coding: utf-8 -*-
import sys
import time
from PyQt5.QtCore import QRunnable, QThreadPool, QObject, pyqtSignal
from PyQt5.QtWidgets import QApplication, QWidget, QPushButton, QTextEdit, QVBoxLayout

class WorkerSignals(QObject):
    finished = pyqtSignal(str)

class Worker(QRunnable):
    def __init__(self, task_id):
        super().__init__()
        self.task_id = task_id
        self.signals = WorkerSignals()
    
    def run(self):
        time.sleep(1)
        self.signals.finished.emit(f'任务 {self.task_id} 完成')

class Example(QWidget):
    def __init__(self):
        super().__init__()
        self.initUI()
    
    def initUI(self):
        self.textEdit = QTextEdit()
        self.btn = QPushButton('启动 5 个任务')
        self.btn.clicked.connect(self.startTasks)
        
        layout = QVBoxLayout()
        layout.addWidget(self.textEdit)
        layout.addWidget(self.btn)
        self.setLayout(layout)
        
        self.threadpool = QThreadPool()
        self.setWindowTitle('线程池示例')
        self.show()
    
    def startTasks(self):
        for i in range(5):
            worker = Worker(i + 1)
            worker.signals.finished.connect(self.onFinished)
            self.threadpool.start(worker)
    
    def onFinished(self, message):
        self.textEdit.append(message)

if __name__ == '__main__':
    app = QApplication(sys.argv)
    ex = Example()
    sys.exit(app.exec_())
```

## 5. 线程安全

### 5.1 使用信号与槽

信号与槽是线程安全的，推荐用于线程间通信。

```python
# 正确做法：使用信号
class Worker(QThread):
    data_ready = pyqtSignal(str)
    
    def run(self):
        result = "处理结果"
        self.data_ready.emit(result)  # 线程安全

# 错误做法：直接修改 UI
class Worker(QThread):
    def __init__(self, text_edit):
        super().__init__()
        self.text_edit = text_edit  # 危险！
    
    def run(self):
        self.text_edit.append("结果")  # 线程不安全！
```

## 6. 多线程最佳实践

| 实践 | 说明 |
|------|------|
| 使用信号与槽 | 线程安全的通信方式 |
| 避免直接访问 UI | 只能在主线程操作 UI |
| 管理线程生命周期 | 确保线程正确退出 |
| 使用线程池 | 对于短任务更高效 |
| 处理异常 | 在线程中捕获异常 |

## 7. 完整示例：文件下载器

```python
# -*- coding: utf-8 -*-
import sys
import time
from PyQt5.QtCore import QThread, pyqtSignal
from PyQt5.QtWidgets import (QApplication, QWidget, QPushButton, 
                             QProgressBar, QVBoxLayout, QLabel)

class DownloadThread(QThread):
    progress = pyqtSignal(int)
    finished = pyqtSignal()
    error = pyqtSignal(str)
    
    def run(self):
        try:
            # 模拟下载
            for i in range(1, 101):
                time.sleep(0.05)
                self.progress.emit(i)
            self.finished.emit()
        except Exception as e:
            self.error.emit(str(e))

class Example(QWidget):
    def __init__(self):
        super().__init__()
        self.initUI()
    
    def initUI(self):
        self.label = QLabel('准备下载')
        self.progressBar = QProgressBar()
        self.btn = QPushButton('开始下载')
        self.btn.clicked.connect(self.startDownload)
        
        layout = QVBoxLayout()
        layout.addWidget(self.label)
        layout.addWidget(self.progressBar)
        layout.addWidget(self.btn)
        self.setLayout(layout)
        
        self.setWindowTitle('文件下载器')
        self.show()
    
    def startDownload(self):
        self.btn.setEnabled(False)
        self.label.setText('下载中...')
        
        self.thread = DownloadThread()
        self.thread.progress.connect(self.progressBar.setValue)
        self.thread.finished.connect(self.onFinished)
        self.thread.error.connect(self.onError)
        self.thread.start()
    
    def onFinished(self):
        self.label.setText('下载完成')
        self.btn.setEnabled(True)
    
    def onError(self, message):
        self.label.setText(f'下载失败: {message}')
        self.btn.setEnabled(True)

if __name__ == '__main__':
    app = QApplication(sys.argv)
    ex = Example()
    sys.exit(app.exec_())
```
