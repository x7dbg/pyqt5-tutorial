# 数据库与样式表

数据库让应用具备数据存储能力，QSS 样式表则让界面更加美观。

## 第一部分：数据库操作

PyQt5 提供了对多种数据库的支持，包括 SQLite、MySQL、PostgreSQL 等。

---

### 1. QtSql 模块简介

#### 1.1 主要类

| 类名 | 作用 |
|------|------|
| `QSqlDatabase` | 管理数据库连接 |
| `QSqlQuery` | 执行 SQL 语句 |
| `QSqlTableModel` | 表格数据模型 |
| `QSqlQueryModel` | 查询数据模型 |

#### 1.2 支持的数据库

- **QSQLITE** - SQLite（内置，无需额外安装）
- **QMYSQL** - MySQL
- **QPSQL** - PostgreSQL
- **QODBC** - ODBC 兼容数据库

### 2. SQLite 数据库操作

SQLite 是轻量级嵌入式数据库，适合桌面应用，无需单独安装服务器。

#### 2.1 连接数据库

```python
# -*- coding: utf-8 -*-
from PyQt5.QtSql import QSqlDatabase, QSqlQuery

# 创建连接
db = QSqlDatabase.addDatabase('QSQLITE')
db.setDatabaseName('test.db')  # 数据库文件名

# 打开连接
if db.open():
    print("数据库连接成功！")
else:
    print(f"连接失败：{db.lastError().text()}")
```

#### 2.2 创建表

```python
# -*- coding: utf-8 -*-
from PyQt5.QtSql import QSqlDatabase, QSqlQuery

db = QSqlDatabase.addDatabase('QSQLITE')
db.setDatabaseName('test.db')

if db.open():
    query = QSqlQuery()
    
    # 创建用户表
    create_sql = """
    CREATE TABLE IF NOT EXISTS users (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        username TEXT NOT NULL,
        email TEXT,
        age INTEGER,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    )
    """
    
    if query.exec(create_sql):
        print("表创建成功！")
    else:
        print(f"创建失败：{query.lastError().text()}")
    
    db.close()
```

#### 2.3 插入数据

```python
# -*- coding: utf-8 -*-
from PyQt5.QtSql import QSqlDatabase, QSqlQuery

db = QSqlDatabase.addDatabase('QSQLITE')
db.setDatabaseName('test.db')

if db.open():
    query = QSqlQuery()
    
    # 方式1：直接执行 SQL
    insert_sql = """
    INSERT INTO users (username, email, age) 
    VALUES ('张三', 'zhangsan@example.com', 25)
    """
    query.exec(insert_sql)
    
    # 方式2：使用占位符（推荐，防止 SQL 注入）
    query.prepare("INSERT INTO users (username, email, age) VALUES (?, ?, ?)")
    query.addBindValue('李四')
    query.addBindValue('lisi@example.com')
    query.addBindValue(30)
    query.exec()
    
    print(f"插入成功，ID：{query.lastInsertId()}")
    db.close()
```

#### 2.4 查询数据

```python
# -*- coding: utf-8 -*-
from PyQt5.QtSql import QSqlDatabase, QSqlQuery

db = QSqlDatabase.addDatabase('QSQLITE')
db.setDatabaseName('test.db')

if db.open():
    query = QSqlQuery()
    
    # 查询所有用户
    query.exec("SELECT id, username, email, age FROM users")
    
    print("ID\t用户名\t\t邮箱\t\t\t年龄")
    print("-" * 60)
    
    while query.next():
        id = query.value(0)
        username = query.value(1)
        email = query.value(2)
        age = query.value(3)
        print(f"{id}\t{username}\t\t{email}\t\t{age}")
    
    # 条件查询
    query.prepare("SELECT * FROM users WHERE age > :age")
    query.bindValue(':age', 25)
    query.exec()
    
    print("\n年龄大于25的用户：")
    while query.next():
        print(query.value('username'))  # 也可以通过字段名访问
    
    db.close()
```

#### 2.5 更新和删除

```python
# -*- coding: utf-8 -*-
from PyQt5.QtSql import QSqlDatabase, QSqlQuery

db = QSqlDatabase.addDatabase('QSQLITE')
db.setDatabaseName('test.db')

if db.open():
    query = QSqlQuery()
    
    # 更新数据
    query.prepare("UPDATE users SET age = :age WHERE username = :name")
    query.bindValue(':age', 26)
    query.bindValue(':name', '张三')
    
    if query.exec():
        print(f"更新了 {query.numRowsAffected()} 条记录")
    
    # 删除数据
    query.prepare("DELETE FROM users WHERE id = :id")
    query.bindValue(':id', 1)
    query.exec()
    
    db.close()
```

### 3. 使用数据模型（Model/View）

#### 3.1 QSqlTableModel

`QSqlTableModel` 提供了高级的表格数据操作，可以直接与 `QTableView` 配合使用。

```python
# -*- coding: utf-8 -*-
import sys
from PyQt5.QtSql import QSqlDatabase, QSqlTableModel
from PyQt5.QtWidgets import (QApplication, QMainWindow, QTableView, 
                             QVBoxLayout, QWidget, QPushButton, 
                             QHBoxLayout, QMessageBox)


class DatabaseApp(QMainWindow):
    def __init__(self):
        super().__init__()
        self.initDB()
        self.initUI()
    
    def initDB(self):
        # 连接数据库
        db = QSqlDatabase.addDatabase('QSQLITE')
        db.setDatabaseName('test.db')
        
        if not db.open():
            QMessageBox.critical(self, "错误", "无法打开数据库")
            return
        
        # 创建模型
        self.model = QSqlTableModel()
        self.model.setTable('users')
        self.model.setEditStrategy(QSqlTableModel.OnFieldChange)  # 自动提交
        self.model.select()  # 加载数据
        
        # 设置表头
        self.model.setHeaderData(0, 1, "ID")
        self.model.setHeaderData(1, 1, "用户名")
        self.model.setHeaderData(2, 1, "邮箱")
        self.model.setHeaderData(3, 1, "年龄")
    
    def initUI(self):
        # 创建表格视图
        self.table_view = QTableView()
        self.table_view.setModel(self.model)
        self.table_view.resizeColumnsToContents()
        
        # 按钮
        btn_add = QPushButton('添加')
        btn_add.clicked.connect(self.add_record)
        
        btn_delete = QPushButton('删除')
        btn_delete.clicked.connect(self.delete_record)
        
        btn_refresh = QPushButton('刷新')
        btn_refresh.clicked.connect(self.model.select)
        
        btn_layout = QHBoxLayout()
        btn_layout.addWidget(btn_add)
        btn_layout.addWidget(btn_delete)
        btn_layout.addWidget(btn_refresh)
        
        # 主布局
        layout = QVBoxLayout()
        layout.addWidget(self.table_view)
        layout.addLayout(btn_layout)
        
        container = QWidget()
        container.setLayout(layout)
        self.setCentralWidget(container)
        
        self.setWindowTitle('数据库浏览器')
        self.resize(600, 400)
    
    def add_record(self):
        row = self.model.rowCount()
        self.model.insertRow(row)
        self.model.setData(self.model.index(row, 1), '新用户')
        self.model.submitAll()
    
    def delete_record(self):
        row = self.table_view.currentIndex().row()
        if row >= 0:
            self.model.removeRow(row)
            self.model.submitAll()
            self.model.select()


if __name__ == '__main__':
    app = QApplication(sys.argv)
    window = DatabaseApp()
    window.show()
    sys.exit(app.exec_())
```

#### 3.2 编辑策略

```python
# 设置编辑策略
self.model.setEditStrategy(QSqlTableModel.OnFieldChange)  # 字段改变时自动提交
self.model.setEditStrategy(QSqlTableModel.OnRowChange)    # 行改变时提交
self.model.setEditStrategy(QSqlTableModel.OnManualSubmit) # 手动提交

# 手动提交示例
self.model.submitAll()  # 提交所有修改
self.model.revertAll()  # 撤销所有修改
```

## 第二部分：QSS 样式表

QSS（Qt Style Sheets）是 Qt 提供的样式表机制，语法类似于 CSS，用于自定义控件的外观。

---

### 4. QSS 基础

#### 4.1 什么是 QSS

QSS 允许我们：

- 自定义控件的颜色、字体、边框
- 设置控件的背景图片
- 定义控件在不同状态下的样式（悬停、按下、禁用等）
- 创建圆角、阴影等视觉效果

#### 4.2 基本语法

```css
/* 选择器 { 属性: 值; } */
QPushButton {
    background-color: #4CAF50;
    color: white;
    border: none;
    padding: 10px 20px;
    font-size: 14px;
}
```

---

### 5. 设置 QSS 的方式

#### 5.1 为单个控件设置样式

```python
# -*- coding: utf-8 -*-
import sys
from PyQt5.QtWidgets import QApplication, QWidget, QPushButton, QVBoxLayout

class Example(QWidget):
    def __init__(self):
        super().__init__()
        self.initUI()
    
    def initUI(self):
        btn = QPushButton('绿色按钮', self)
        btn.move(50, 50)
        
        # 为单个控件设置样式
        btn.setStyleSheet("""
            QPushButton {
                background-color: #4CAF50;
                color: white;
                border: none;
                padding: 10px 20px;
                border-radius: 5px;
            }
            QPushButton:hover {
                background-color: #45a049;
            }
        """)
        
        self.setWindowTitle('QSS 示例')
        self.resize(300, 200)

if __name__ == '__main__':
    app = QApplication(sys.argv)
    ex = Example()
    ex.show()
    sys.exit(app.exec_())
```

#### 5.2 为整个应用设置样式

```python
# -*- coding: utf-8 -*-
import sys
from PyQt5.QtWidgets import QApplication, QWidget, QPushButton, QVBoxLayout, QLabel

app = QApplication(sys.argv)

# 为整个应用设置样式表
app.setStyleSheet("""
    QPushButton {
        background-color: #4CAF50;
        color: white;
        border: none;
        padding: 10px 20px;
        border-radius: 5px;
    }
    
    QPushButton:hover {
        background-color: #45a049;
    }
    
    QLabel {
        color: #333;
        font-size: 14px;
    }
    
    QWidget {
        background-color: #f5f5f5;
    }
""")

window = QWidget()
layout = QVBoxLayout()
layout.addWidget(QLabel('应用级样式表示例'))
layout.addWidget(QPushButton('按钮1'))
layout.addWidget(QPushButton('按钮2'))
window.setLayout(layout)
window.setWindowTitle('全局样式')
window.show()

sys.exit(app.exec_())
```

#### 5.3 从文件加载样式

```python
# -*- coding: utf-8 -*-
def load_style_sheet(file_path):
    """从文件加载样式表"""
    with open(file_path, 'r', encoding='utf-8') as f:
        return f.read()

# 使用
app.setStyleSheet(load_style_sheet('style.qss'))
```

### 8. 完整示例：美化登录界面

```python
# -*- coding: utf-8 -*-
import sys
from PyQt5.QtCore import Qt
from PyQt5.QtWidgets import (QApplication, QWidget, QVBoxLayout, 
                             QHBoxLayout, QLabel, QLineEdit, 
                             QPushButton, QFormLayout)

class LoginWindow(QWidget):
    def __init__(self):
        super().__init__()
        self.initUI()
    
    def initUI(self):
        # 标题
        title = QLabel('用户登录')
        title.setObjectName('title')
        title.setAlignment(Qt.AlignCenter)
        
        # 表单
        self.username = QLineEdit()
        self.username.setPlaceholderText('请输入用户名')
        self.username.setObjectName('username')
        
        self.password = QLineEdit()
        self.password.setPlaceholderText('请输入密码')
        self.password.setEchoMode(QLineEdit.Password)
        self.password.setObjectName('password')
        
        form_layout = QFormLayout()
        form_layout.addRow('用户名:', self.username)
        form_layout.addRow('密码:', self.password)
        
        # 按钮
        self.login_btn = QPushButton('登录')
        self.login_btn.setObjectName('loginBtn')
        self.login_btn.clicked.connect(self.on_login)
        
        self.cancel_btn = QPushButton('取消')
        self.cancel_btn.setObjectName('cancelBtn')
        self.cancel_btn.clicked.connect(self.close)
        
        btn_layout = QHBoxLayout()
        btn_layout.addWidget(self.login_btn)
        btn_layout.addWidget(self.cancel_btn)
        
        # 主布局
        layout = QVBoxLayout()
        layout.addWidget(title)
        layout.addLayout(form_layout)
        layout.addLayout(btn_layout)
        self.setLayout(layout)
        
        # 设置样式
        self.setStyleSheet("""
            QWidget {
                background-color: #f0f2f5;
            }
            
            QLabel#title {
                color: #1890ff;
                font-size: 24px;
                font-weight: bold;
                padding: 20px;
            }
            
            QLineEdit {
                background-color: white;
                border: 1px solid #d9d9d9;
                border-radius: 4px;
                padding: 8px;
                font-size: 14px;
            }
            
            QLineEdit:focus {
                border-color: #1890ff;
            }
            
            QPushButton#loginBtn {
                background-color: #1890ff;
                color: white;
                border: none;
                border-radius: 4px;
                padding: 10px;
                font-size: 16px;
                font-weight: bold;
            }
            
            QPushButton#loginBtn:hover {
                background-color: #40a9ff;
            }
            
            QPushButton#loginBtn:pressed {
                background-color: #096dd9;
            }
            
            QPushButton#cancelBtn {
                background-color: white;
                color: #666;
                border: 1px solid #d9d9d9;
                border-radius: 4px;
                padding: 10px;
                font-size: 16px;
            }
            
            QPushButton#cancelBtn:hover {
                border-color: #1890ff;
                color: #1890ff;
            }
        """)
        
        self.setWindowTitle('登录')
        self.resize(350, 250)
    
    def on_login(self):
        username = self.username.text()
        password = self.password.text()
        print(f'登录: {username}')

if __name__ == '__main__':
    app = QApplication(sys.argv)
    window = LoginWindow()
    window.show()
    sys.exit(app.exec_())
```
